#include <stdio.h>


int funcao(int cpf1, int cpf2, int cpf3, int cpf4, int cpf5, int cpf6, int cpf7, int cpf8, int cpf9) {
    int R1 = cpf1 * 10;	
    int R2 = cpf2 * 9;
    int R3 = cpf3 * 8;
    int R4 = cpf4 * 7;
    int R5 = cpf5 * 6;
    int R6 = cpf6 * 5;
    int R7 = cpf7 * 4;
    int R8 = cpf8 * 3;
    int R9 = cpf9 * 2;
    int soma = R1 + R2 + R3 + R4 + R5 + R6 + R7 + R8 + R9;
    int resto = soma % 11;
    int primeiroDigito;
    
    if (resto < 2){
        primeiroDigito = 0;
    } else {
        primeiroDigito = 11 - resto;
    }

    printf("\nPrimeiro digito verificador: %d\n", primeiroDigito);
    
    return primeiroDigito;
}

    S6 = cpf6 * 6;
    S7 = cpf7 * 5;

    S10 = primeiroDigito * 2;

    int soma_2, resto_2, segundoDigito;
    soma_2 = S1 + S2 + S3 + S4 + S5 + S6 + S7 + S8 + S9 + S10;
    resto_2 = soma_2 % 11;

    if (resto_2 < 2){
        segundoDigito = 0;
    }
    if (resto_2 >= 2){
        segundoDigito = 11 - resto_2;
    }

    printf("Segundo digito verificador: %d\n", segundoDigito);

    // Exibindo CPF completo
    printf("\nCPF completo: %d%d%d.%d%d%d.%d%d%d-%d%d\3n",
           cpf1, cpf2, cpf3,
           cpf4, cpf5, cpf6,
           cpf7, cpf8, cpf9,
           primeiroDigito, segundoDigito);
           
    return segundoDigito;
}
